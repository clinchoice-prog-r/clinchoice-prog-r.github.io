Plot sample Student-t distribution

x <- seq(-5, 5, by = 0.01)
y <- dt(x, df = Inf)
y2 <- dt(x, df = 10)
y3 <- dt(x, df = 5)
y4 <- dt(x, df = 3)
y5 <- dt(x, df = 1)
plot(y, type = "l", main = "t-distribution",col = "Red")
lines(y2,col = "Orange",)
lines(y3,col = "Green")
lines(y4,col = "Blue")
lines(y5,col = "Purple")
legend(800, 0.4, legend=c("df = Inf", "df = 10", "df = 5","df = 3", "df = 1"),
       col=c("Red", "Orange", "Green", "Blue", "Purple"), lty=1)
